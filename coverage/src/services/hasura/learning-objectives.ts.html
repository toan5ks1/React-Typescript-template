import { ProviderTypes } from "src/common/constants/enum";
import { AppPagination, ListQuery, GqlSort } from "src/services/service-types";
import { RaSort } from "src/typings/react-admin";
 
import { learningObjectivesQueriesBob } from "../bob/learning-objectives-service-bob";
import learningObjectivesService from "../bob/learning-objectives-service-bob/learning-object.mutation";
import courseModifierServiceEureka from "../eureka/course-modifier-eureka";
import { NsEurekaCourseModifierService } from "../eureka/course-modifier-eureka/types";
import {
    calcGqlPagination,
    getEmptyResponse,
    toGqlQuerySorts,
    createEmptyResponse,
} from "../utils";
import courseServiceYasuo from "../yasuo/course-service-yasuo";
 
type Params =
    | {
          type: ProviderTypes.UPDATE;
          payload: Parameters<typeof learningObjectivesService.upsertLOs>[0];
      }
    | {
          type: ProviderTypes.CREATE;
          payload: Parameters<typeof learningObjectivesService.upsertLOs>[0];
      }
    | {
          type: ProviderTypes.DELETE;
          payload: Parameters<typeof courseServiceYasuo.deleteLOs>[0];
      }
    | {
          type: ProviderTypes.ONE;
          payload: Parameters<typeof learningObjectivesQueriesBob.getOne>[0] & {
              id: string;
          } & ListQuery<
                  Parameters<typeof learningObjectivesQueriesBob.getOne>[0] & {
                      entityId: string;
                  }
              >;
      }
    | {
          type: ProviderTypes.MANY;
          payload: Parameters<typeof learningObjectivesQueriesBob.getMany>[0] & {
              filter?: { lo_ids: string[] };
              ids?: string[];
          };
      }
    | {
          type: ProviderTypes.LIST_WITH_FILTER;
          payload: Parameters<typeof learningObjectivesQueriesBob.getByTopicIds>[0] &
              ListQuery<Parameters<typeof learningObjectivesQueriesBob.getByTopicIds>[0]>;
      }
    | {
          type: ProviderTypes.LIST;
          payload: ListQuery<
              Parameters<typeof learningObjectivesQueriesBob.getList>[0],
              RaSort | RaSort[] | GqlSort,
              AppPagination
          >;
      }
    | {
          type: ProviderTypes.DISPLAY_ORDER;
          payload: { filter: Parameters<typeof learningObjectivesQueriesBob.getDisplayOrder>[0] };
      }
    | {
          type: ProviderTypes.UPDATE_DISPLAY_ORDER;
          payload: NsEurekaCourseModifierService.UpdateDisplayOrdersOfLOsAndAssignments;
      };
 
export default (params: Params) => {
    switch (params.type) {
        case ProviderTypes.UPDATE:
        case ProviderTypes.CREATE: {
            return learningObjectivesService.upsertLOs(params.payload);
        }
 
        case ProviderTypes.DELETE:
            return courseServiceYasuo.deleteLOs(params.payload);
 
        case ProviderTypes.MANY: {
            const { payload } = params;
 
            return learningObjectivesQueriesBob.getMany({
                lo_id: payload.lo_id || payload.ids || payload.filter?.lo_ids,
            });
        }
 
        case ProviderTypes.LIST: {
            const { pagination, filter = {}, sort } = params.payload;
            const { limit, offset } = calcGqlPagination(pagination);
            const { lo_id, topic_id, name, type, grade, country, subject } = filter;
 
            return learningObjectivesQueriesBob.getList({
                limit,
                offset,
                lo_id,
                topic_id,
                name,
                type,
                grade,
                country,
                subject,
                order_by: toGqlQuerySorts(sort),
            });
        }
        case ProviderTypes.LIST_WITH_FILTER: {
            const { topic_id, filter } = params.payload;
 
            return learningObjectivesQueriesBob.getByTopicIds({
                topic_id: topic_id || filter?.topic_id,
            });
        }
        case ProviderTypes.ONE: {
            const { lo_id, id, filter } = params.payload;
 
            const finalId = lo_id || id || filter?.lo_id || filter?.entityId;
 
            if (!finalId) return createEmptyResponse({ data: null });
 
            return learningObjectivesQueriesBob.getOne({
                lo_id: finalId,
            });
        }
 
        case ProviderTypes.DISPLAY_ORDER: {
            const { topic_id } = params.payload.filter;
 
            return learningObjectivesQueriesBob.getDisplayOrder({ topic_id });
        }
 
        case ProviderTypes.UPDATE_DISPLAY_ORDER: {
            return courseModifierServiceEureka.updateDisplayOrdersOfLOsAndAssignments(
                params.payload
            );
        }
 
        default: {
            return getEmptyResponse();
        }
    }
};
