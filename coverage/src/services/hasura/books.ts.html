import { ProviderTypes } from "src/common/constants/enum";
import { RaSort } from "src/typings/react-admin";
 
import { bookQueriesBob } from "../bob/book-service-bob";
import courseModifierServiceEureka from "../eureka/course-modifier-eureka";
import { AppPagination, GqlPagination, ListQuery } from "../service-types";
import { calcGqlPagination, getEmptyResponse } from "../utils";
import courseServiceYasuo from "../yasuo/course-service-yasuo";
 
type Params =
    | {
          type: ProviderTypes.TITLE;
          payload: Parameters<typeof bookQueriesBob.getTitle>[0] & { id?: string } & ListQuery<

              >;
      }
    | {

          payload: Parameters<typeof bookQueriesBob.getOne>[0] & { id?: string } & ListQuery<
                  Parameters<typeof bookQueriesBob.getOne>[0] & {
                      book_id: string;
                      entityId: string;
                  }

      }
    | {
          type: ProviderTypes.MANY;
          payload: Parameters<typeof bookQueriesBob.getMany>[0] & { ids?: string[] } & ListQuery<{
                  ids: string;
              }>;
      }
    | {
          type: ProviderTypes.MANY_REFERENCE;
          payload: ListQuery<

const hasuraBooks = (params: Params) => {
    switch (params.type) {
        case ProviderTypes.TITLE: {
            const { id, book_id, filter } = params.payload;
            const finalId = book_id || id || filter?.book_id || "";
 
            return bookQueriesBob.getTitle({ book_id: finalId });
        }
        case ProviderTypes.ONE: {
            const { id, book_id, filter } = params.payload;
            const finalId = book_id || id || filter?.book_id || filter?.entityId || "";
 
            return bookQueriesBob.getOne({ book_id: finalId });
        }
        case ProviderTypes.UPDATE:
        case ProviderTypes.CREATE: {
            return courseServiceYasuo.upsertBooks(params.payload);
        }
        case ProviderTypes.DELETE: {
            const { id, bookId } = params.payload;
            return courseServiceYasuo.deleteBook({ bookId: bookId || id! });
        }
        case ProviderTypes.REMOVE_BOOK: {
            return courseServiceYasuo.removeBooksFromCourse(params.payload);
        }
        case ProviderTypes.DUPLICATE_BOOK: {
            return courseModifierServiceEureka.duplicateBook(params.payload);
        }
 
        case ProviderTypes.MANY: {
            const { ids, book_id, filter } = params.payload;
            return bookQueriesBob.getMany({ book_id: book_id || ids! || filter?.ids });
        }
 
        case ProviderTypes.MANY_REFERENCE: {
            const { pagination, filter } = params.payload;
            const { limit, offset } = calcGqlPagination(pagination);
 
            return bookQueriesBob.getManyReference({
                name: filter?.name,
                limit,
                offset,
            });
        }
 
        case ProviderTypes.LIST: {
            const { pagination, filter } = params.payload;
            if (!filter || !pagination) return getEmptyResponse();
 
            const { limit, offset } = pagination;
 
            return bookQueriesBob.getList({
                name: filter?.name,
                limit,
                offset,
            });
        }
 
        case ProviderTypes.LIST_WITH_FILTER: {
            const { pagination, filter } = params.payload;
            const { course_id, name } = filter || {};
            const { limit, offset } = calcGqlPagination(pagination);
 
            return bookQueriesBob.getListNotInCourseId({
                course_id,
                name,
                limit,
                offset,
            });
        }
 
        default: {
            return getEmptyResponse();
        }
    }
};
 
export default hasuraBooks;
